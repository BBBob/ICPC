#define bf(x) ((x)<<1)
#define bt(x) ((x)<<1|1)
int ind[N], id[N];
int t[M], nt[M];
int dfn[N], low[N];
int s[N];
bool vis[N];
int l;
int n;
int cnt, num, idn;
int a[N], b[N], c[N];

void add(int i, int j)
{
    cnt++;
    t[cnt] = j;
    nt[cnt] = ind[i];
    ind[i] = cnt;
}

void tarjan(int x)
{
    num++;
    dfn[x] = low[x] = num;
    s[++l] = x;
    vis[x] = true;
    for(int k = ind[x]; k != -1; k = nt[k])
    {
        if(dfn[t[k]] == 0) tarjan(t[k]);
        if(vis[t[k]]) low[x] = min(low[x], low[t[k]]);
    }
    if(dfn[x] == low[x])
    {
        idn++;
        while(true)
        {
            int tmp = s[l--];
            id[tmp] = idn;
            vis[tmp] = false;
            if(tmp == x) break;
        }
    }
}

bool chk(int M)
{
    for(int i = 0; i < 2 * n; i++) ind[i] = -1;
    cnt = 0;
	/*
	* ADD Edge
	*/
    memset(vis, 0, sizeof vis);
    num = 0;
    idn = 0;
    l = 0;
    memset(dfn, 0, sizeof dfn);
    memset(low, 0, sizeof low);
    memset(id, 0, sizeof id);
    for(int i = 0; i < 2 * n; i++)
    {
        if(dfn[i] == 0) tarjan(i);
    }
    for(int i = 0; i < 2 * n; i++)
    {
        if(id[i] == id[i ^ 1]) return false;
    }
    return true;
}
