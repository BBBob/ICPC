\subsection{欧拉函数}
\paragraph{}
对正整数n，欧拉函数是少于或等于n的数中与n互质的数的数目
    \begin{lstlisting}[language=C++]
int prime[N];
int phi[N];
bool is_prime[N];

void get_phi()
{
    int i, j, k;
    k = 0;
    for(i = 2; i < N; i++)
    {
        if(is_prime[i] == false)
        {
            prime[k++] = i;
            phi[i] = i - 1;
        }
        for(j = 0; j < k && i * prime[j] < N; j++)
        {
            is_prime[ i * prime[j] ] = true;
            if( i % prime[j] == 0)
            {
                phi[ i * prime[j] ] = phi[i] * prime[j];
                break;
            }
            else
            {
                phi[ i * prime[j] ] = phi[i] * ( prime[j] - 1 );
            }
        }
    }
}
\end{lstlisting}
