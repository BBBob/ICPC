void getNext()
{    
	int j,k;
    j = 0;
    k = -1;
    next[0] = -1;    
	while(j < m)
    {        
		if(k == -1 || b[j] == b[k])
        next[++j] = ++k;        
		else k = next[k];
    }    
}  
// get First Position
int KMP_Index()
{    
	int i = 0, j = 0;
    getNext();    
    while(i < n && j < m)
    {        
		if(j == -1||a[i] == b[j])
        {
            i++;
            j++;
        }    
        else j = next[j];        
    }    
    if(j == m) return i - m + 1;    
	else return -1;
}
