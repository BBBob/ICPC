int f[N][POW];
int mm[N];
for(int j = 0; j < POW; j++)
    for(int i = 0; i < n; i++)
    if(i + (1 << (j + 1)) <= n)
    {
        f[i][j + 1] = max(f[i][j], f[i + (1 << j)][j]);
    }
mm[0] = -1;
for(int i = 0; i < N; i++) {
	if((i & (i - 1)) == 0) mm[i] = mm[i - 1] + 1;
	else mm[i] = mm[i - 1];
}
int k = mm[r - l + 1];
ans = max(ans, max(f[l][k], f[r - qpow(2, k) + 1][k]));
