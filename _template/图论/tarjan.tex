\subsection{2-SAT}
    \begin{lstlisting}[language=C++]
int ind[N];
int t[M], nt[M];
int dfn[N], low[N];
bool vis[N];
int n;
int cnt, num, idn;
stack<int>s;

void add(int i, int j)
{
    cnt++;
    t[cnt] = j;
    nt[cnt] = ind[i];
    ind[i] = cnt;
}

void tarjan(int x) {
	if(dfn[x]) return;
	low[x] = dfn[x] = ++num;
	vis[x] = true;
	s.push(x);
	for(int k = ind[x]; k != -1; k = nt[k]) {
		tarjan(to[k]);
		if(vis[to[k]]) low[x] = min(low[x], low[to[k]]);
	}
	if(dfn[x] == low[x]) {
		while(true) {
			int p = s.top();
			s.pop();
		}
	}
}
    \end{lstlisting}
