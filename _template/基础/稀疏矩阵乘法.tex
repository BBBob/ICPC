\subsection{稀疏矩阵乘法}
    \begin{lstlisting}[language=C++]
int n; //`矩阵大小`
void matrixMul(LL a[][],LL b[][],LL c[][])
{
	/*
	* `要清空c`
	*/
    for(int i = 0;i < n; i++)
    {
        for(int j = 0;j < n; j++)
        {
            if(0 == a[i][j])continue;
            for(int k = 0; k < n; k++)
            c[i][k] += a[i][j] * b[j][k];
        }
    }
}
    \end{lstlisting}
